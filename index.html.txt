<script>

(function(){
  const LS_HISTORY = "alvi_upload_history";
  const LS_ENDPOINT = "alvi_upload_endpoint";
  const LS_TOKEN = "alvi_upload_token";
  const MAX_SIZE = 20 * 1024 * 1024;
  const ALLOWED = ["image/png","image/jpeg","image/webp","image/gif","audio/mpeg"];

  const dropzone = document.getElementById("dropzone");
  const fileInput = document.getElementById("fileInput");
  const pipeline = document.getElementById("pipeline");
  const pFilename = document.getElementById("pFilename");
  const pPercent = document.getElementById("pPercent");
  const progressFill = document.getElementById("progressFill");
  const resultBox = document.getElementById("resultBox");
  const resultUrl = document.getElementById("resultUrl");
  const copyBtn = document.getElementById("copyBtn");
  const errorBox = document.getElementById("errorBox");
  const historyList = document.getElementById("historyList");
  const clearHistoryBtn = document.getElementById("clearHistoryBtn");

  const gearBtn = document.getElementById("gearBtn");
  const settingsModal = document.getElementById("settingsModal");
  const endpointInput = document.getElementById("endpointInput");
  const tokenInput = document.getElementById("tokenInput");
  const saveSettingsBtn = document.getElementById("saveSettingsBtn");
  const cancelSettingsBtn = document.getElementById("cancelSettingsBtn");

  const stages = [1,2,3].map(n => ({ el: document.getElementById("stage"+n) }));
  const lines = [1,2].map(n => document.getElementById("line"+n));

  function getConfig(){
    return {
      endpoint: localStorage.getItem(LS_ENDPOINT) || "",
      token: localStorage.getItem(LS_TOKEN) || "",
    };
  }
  function isConfigured(){ const c = getConfig(); return !!(c.endpoint && c.token); }

  function openSettings(){
    const c = getConfig();
    endpointInput.value = c.endpoint;
    tokenInput.value = c.token;
    settingsModal.classList.add("show");
  }
  function closeSettings(){ settingsModal.classList.remove("show"); }

  gearBtn.addEventListener("click", openSettings);
  cancelSettingsBtn.addEventListener("click", closeSettings);
  saveSettingsBtn.addEventListener("click", () => {
    localStorage.setItem(LS_ENDPOINT, endpointInput.value.trim());
    localStorage.setItem(LS_TOKEN, tokenInput.value.trim());
    closeSettings();
  });

  dropzone.addEventListener("click", () => fileInput.click());
  ["dragenter","dragover"].forEach(ev => dropzone.addEventListener(ev, e => { e.preventDefault(); dropzone.classList.add("drag"); }));
  ["dragleave","drop"].forEach(ev => dropzone.addEventListener(ev, e => { e.preventDefault(); dropzone.classList.remove("drag"); }));
  dropzone.addEventListener("drop", e => { const f = e.dataTransfer.files[0]; if (f) handleFile(f); });
  fileInput.addEventListener("change", e => { const f = e.target.files[0]; if (f) handleFile(f); fileInput.value = ""; });

  function resetPipeline(){
    stages.forEach(s => s.el.classList.remove("active","done","error"));
    lines.forEach(l => l.classList.remove("done"));
    progressFill.style.width = "0%";
    pPercent.textContent = "0%";
    resultBox.classList.remove("show");
    errorBox.classList.remove("show");
    errorBox.textContent = "";
  }

  function setStage(idx, state){
    stages[idx].el.classList.remove("active","done","error");
    stages[idx].el.classList.add(state);
    if (idx > 0 && state === "done") lines[idx-1].classList.add("done");
  }

  function handleFile(file){
    if (!isConfigured()){ openSettings(); return; }
    if (!ALLOWED.includes(file.type)){ showError("不支持的文件类型"); return; }
    if (file.size > MAX_SIZE){ showError("超过 20MB (" + (file.size/1024/1024).toFixed(1) + "MB)"); return; }
    upload(file);
  }

  function showError(msg){
    pipeline.style.display = "block";
    errorBox.textContent = msg;
    errorBox.classList.add("show");
  }

  function upload(file){
    resetPipeline();
    pipeline.style.display = "block";
    pFilename.textContent = file.name;
    setStage(0, "active");

    const cfg = getConfig();
    const xhr = new XMLHttpRequest();
    const formData = new FormData();
    formData.append("file", file);

    xhr.upload.onprogress = (e) => {
      if (!e.lengthComputable) return;
      const pct = Math.round((e.loaded / e.total) * 100);
      progressFill.style.width = pct + "%";
      pPercent.textContent = pct + "%";
      if (pct >= 100){ setStage(0, "done"); setStage(1, "active"); }
    };

    xhr.onload = () => {
      let data;
      try { data = JSON.parse(xhr.responseText); } catch(e){ data = null; }
      if (xhr.status === 200 && data && data.url){
        setStage(1, "done"); setStage(2, "done");
        resultUrl.value = data.url;
        resultBox.classList.add("show");
        addHistory({ name: file.name, url: data.url, path: data.path, time: Date.now(), type: file.type });
      } else {
        setStage(1, "error");
        showError((data && (data.error || data.detail)) || ("上传失败 (HTTP " + xhr.status + ")"));
      }
    };

    xhr.onerror = () => { setStage(1, "error"); showError("网络错误，请检查 Worker 地址"); };
    xhr.open("POST", cfg.endpoint);
    xhr.setRequestHeader("X-Upload-Token", cfg.token);
    xhr.send(formData);
  }

  
copyBtn.addEventListener("click", () => {
    resultUrl.select(); resultUrl.setSelectionRange(0, 99999);
    navigator.clipboard.writeText(resultUrl.value).then(() => {
      copyBtn.textContent = "已复制"; copyBtn.classList.add("copied");
      setTimeout(() => { copyBtn.textContent = "复制"; copyBtn.classList.remove("copied"); }, 1500);
    });
  });

  function getHistory(){ try { return JSON.parse(localStorage.getItem(LS_HISTORY) || "[]"); } catch(e){ return []; } }
  function saveHistory(list){ localStorage.setItem(LS_HISTORY, JSON.stringify(list.slice(0, 100))); }
  function addHistory(item){ const list = getHistory(); list.unshift(item); saveHistory(list); renderHistory(); }  function relTime(ts){
    const diff = Date.now() - ts; const min = Math.floor(diff / 60000);
    if (min < 1) return "刚刚"; if (min < 60) return min + " 分钟前";
    const hr = Math.floor(min / 60); if (hr < 24) return hr + " 小时前";
    return Math.floor(hr / 24) + " 天前";
  }
  function iconFor(type){ if (type === "audio/mpeg") return "♪"; if (type === "image/gif") return "GIF"; return "🖼"; }
  function renderHistory(){
    const list = getHistory();
    if (list.length === 0){ historyList.innerHTML = '<div class="empty-hint">暂无记录</div>'; return; }
    historyList.innerHTML = list.map((item, i) => {
      const isImg = item.type && item.type.startsWith("image/");
      const thumb = isImg ? '< img src="'+ item.url +'" loading="lazy" onerror="this.parentElement.textContent=\'?\'" />' : iconFor(item.type);
      return (
        '<div class="history-item">' +
          '<div class="history-thumb">'+ thumb +'</div>' +
          '<div class="history-info"><div class="history-name">'+ escapeHtml(item.name) +'</div><div class="history-time">'+ relTime(item.time) +'</div></div>' +
          '<button class="history-copy" data-idx="'+ i +'">⧉</button>' +
        '</div>'
      );
    }).join("");
    historyList.querySelectorAll(".history-copy").forEach(btn => {
      btn.addEventListener("click", () => {
        const idx = parseInt(btn.getAttribute("data-idx"), 10);
        const item = getHistory()[idx];
        if (item) navigator.clipboard.writeText(item.url).then(() => { btn.textContent = "✓"; setTimeout(() => btn.textContent = "⧉", 1200); });
      });
    });
  }
  function escapeHtml(str){ const d = document.createElement("div"); d.textContent = str; return d.innerHTML; }

  clearHistoryBtn.addEventListener("click", () => {
    if (confirm("仅清除本机记录，GitHub 文件不受影响。确定？")){ localStorage.removeItem(LS_HISTORY); renderHistory(); }
  });

  renderHistory();
  if (!isConfigured()) setTimeout(openSettings, 300);
})();
</script>
